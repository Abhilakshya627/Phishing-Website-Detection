// content.js - Runs in the context of the webpage
const knownPhishingSites = ["phishingsite.com", "malicioussite.net"];

function checkForPhishing() {
  const currentUrl = window.location.hostname;
  if (knownPhishingSites.includes(currentUrl)) {
    alert("Warning: This site is known for phishing. Proceed with caution.");
  }
}

checkForPhishing();
